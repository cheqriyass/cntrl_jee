52       function submitForm(action) { 
53           var Form = document.forms[0]; 
54           var selected = $("#prixParJour option:selected"); 
55           var x = selected.text(); 
56           selected.remove(); 
57            
58           $('#prixParJour').append($("<option></option>").attr("value",x).text(x)); 
59           $("#prixParJour").val(x); 
60           Form.action = action; 
61           Form.submit(); 
62       }

64       function calcTotal() { 
65           var nbrJours = document.getElementById("nbrJours"); 
66           var prixParJour = document.getElementById("prixParJour"); 
67           var prixTot = document.getElementById("prixTot"); 
68           if (nbrJours.value != "") 
69               prixTot.value = nbrJours.value * prixParJour.value; 
70       }
